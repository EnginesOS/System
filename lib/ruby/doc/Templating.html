  name = match.sub!(/_Blueprint\(/,"")
  name.sub!(/[\)]/,"")
  p :getting_blueprint_value_for
  p name
  val =""
  
   keys = name.split(',')
   hash = @builder_public.blueprint
   keys.each do |key|
     if key == nil || key.length < 1
       break
     end
     p :key
     p key
     val = hash[key.to_sym]
     p :val
     p val     
     if val != nil
       hash=val
     end
   end     
  
  p :got_val
  p val
  
  return val
  
rescue Exception=>e
    SystemUtils.log_exception(e)

end
