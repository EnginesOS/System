def cache_update_ts(container, ts) 
  ident =  get_ident(container)
  name_key = ident.to_sym
  return false unless  @engines_conf_cache.key?(name_key) && ! @engines_conf_cache[name_key].nil?
  @engines_conf_cache[name_key][:ts] = ts
   return true
end
